# payload_binder
use this tool for binding payload to any apk and image
# HOW TO USE
git clone https://github.com/mhdashik0/payload_binder
cd payload_binder
for apk binder
chmod +x apk-binder.sh
bash apk-binder.sh
for image binder
chmod +x image-binder.sh
bash image-binder.sh
